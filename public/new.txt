upstream unicorn_domain_for_sale {
  server unix:/apps/domain_for_sale/unicorn.sock fail_timeout=0;
}

server {
  listen 172.31.35.52:80 default deferred;
  root /apps/domain_for_sale;
  server_name thetwilio.*;

  location ^~ /domain_for_sale/assets/ {
    gzip_static on;
    expires max;
    add_header Cache-Control public;
  }

 try_files $uri/index.html $uri @unicorn_domain_proxy;
  location @unicorn_domain_proxy {
    proxy_set_header X-Forwarded-For $proxy_add_x_forwarded_for;
    proxy_set_header Host $http_host;
    proxy_redirect off;
    #proxy_pass http://unicorn_domain_for_sale;
    proxy_pass        http://172.31.35.52:4000;
    proxy_set_header X-Forwarded-Proto $scheme;
  }

  error_page 500 502 503 504 /500.html;
  client_max_body_size 4G;
  keepalive_timeout 600;
}
